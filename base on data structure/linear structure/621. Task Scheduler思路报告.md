class Solution {
    public int leastInterval(char[] tasks, int n) {
        int[] freq = new int[26];
        int res = tasks.length;

        for (char task: tasks) {
            freq[task - 'A']++;
        }

        int count = 1;
        int maxFreq = 0;
        for (int i = 0; i < 26; i++) {
            int curFreq = freq[i];

            if (curFreq > maxFreq) {
                maxFreq = curFreq;
                count = 1;
            } else if(curFreq == maxFreq) {
                count++;
            }
        }
        int least = (maxFreq - 1) * (n + 1) + count;

        return res > least ? res : least;
    }
}
