class Solution {
    public TreeNode convertBST(TreeNode root) {
        if (root == null) {
            return root;
        }

        int[] rightSum = new int[1];
        inOrder(root, rightSum);

        return root;
    }

    public void inOrder(TreeNode root, int[] rightSum) {
        if (root == null) {
            return;
        }

        inOrder(root.right, rightSum);

        root.val += rightSum[0];
        rightSum[0] = root.val;

        inOrder(root.left, rightSum);
    }
}
